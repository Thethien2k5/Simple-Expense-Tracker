<script>
    // Custom toggle logic for checkboxes to handle visual dot state
    document.querySelectorAll('.toggle-checkbox').forEach(toggle => {
      toggle.addEventListener('change', function() {
        const dot = this.parentElement.querySelector('.toggle-dot');
        const container = this.closest('.bg-surface-container-low');
        if(this.checked) {
           dot.classList.remove('bg-surface-variant');
           dot.classList.add('bg-white');
           if(container) container.classList.remove('opacity-70');
        } else {
           dot.classList.remove('bg-white');
           dot.classList.add('bg-surface-variant');
           if(container) container.classList.add('opacity-70');
        }
      });
    });
  </script>
