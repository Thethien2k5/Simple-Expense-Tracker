<script>
    function toggleModal(modalId, show) {
        const modal = document.getElementById(modalId);
        if (show) {
            modal.classList.remove('hidden');
            document.body.style.overflow = 'hidden';
        } else {
            modal.classList.add('hidden');
            document.body.style.overflow = 'auto';
        }
    }

    function switchChart(type, btn) {
        // Hide all charts
        document.querySelectorAll('.chart-content').forEach(c => c.classList.add('hidden'));
        document.querySelectorAll('.chart-content').forEach(c => c.classList.remove('flex'));
        
        // Show selected chart
        const selected = document.getElementById('chart-' + type);
        selected.classList.remove('hidden');
        if (type === 'tron' || type === 'tru') {
            selected.classList.add('flex');
        }

        // Update tabs style
        document.querySelectorAll('.chart-tab').forEach(t => {
            t.classList.remove('bg-primary', 'text-on-primary');
            t.classList.add('text-on-surface-variant');
        });
        btn.classList.add('bg-primary', 'text-on-primary');
        btn.classList.remove('text-on-surface-variant');
    }
</script>
